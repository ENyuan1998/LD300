// ==================== 設定區 ==================== const SPREADSHEET_ID = '1O5_nN6zzkQZT9AZgsAK3jtkNY1YgXjLirFBYR5JsoPA'; const NOTIFICATION_EMAIL = 'your-email@example.com'; // ⚠️ 請改成您的信箱 // ==================== 主要函數 ==================== function doPost(e) { try { let data; if (e.postData && e.postData.contents) { data = JSON.parse(e.postData.contents); } else if (e.parameter) { data = e.parameter; } else { throw new Error('無法解析請求資料'); } const result = processOrder(data); return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch (error) { Logger.log('錯誤: ' + error.toString()); return ContentService .createTextOutput(JSON.stringify({ success: false, message: '訂單處理失敗：' + error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function processOrder(data) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const orderSheet = ss.getSheetByName('訂單資料'); const customerSheet = ss.getSheetByName('客戶資料'); if (!orderSheet || !customerSheet) { throw new Error('找不到必要的工作表'); } const orderNumber = generateOrderNumber(); const orderDate = new Date(); const orderDateStr = Utilities.formatDate(orderDate, 'GMT+8', 'yyyy/MM/dd'); const orderTimeStr = Utilities.formatDate(orderDate, 'GMT+8', 'HH:mm:ss'); // 處理服務項目明細 let serviceItemsText = ''; if (data.serviceItems && Array.isArray(data.serviceItems)) { serviceItemsText = data.serviceItems.map(item => `${item.name} × ${item.quantity} = NT$ ${item.price * item.quantity}` ).join('\n'); } // 寫入訂單資料 orderSheet.appendRow([ orderNumber, orderDateStr, orderTimeStr, data.serviceType || '', data.name || '', data.phone || '', data.bookingDate || '', data.bookingTime || '', data.startAddress || '', data.endAddress || '', serviceItemsText, // 服務項目明細 data.requirements || '', data.totalAmount || 0, '待處理' ]); updateCustomerData(customerSheet, data); try { sendOrderNotification(orderNumber, data, serviceItemsText); } catch (emailError) { Logger.log('郵件發送失敗: ' + emailError.toString()); } return { success: true, orderNumber: orderNumber, message: '訂單已成功送出！' }; } function generateOrderNumber() { const date = new Date(); const dateStr = Utilities.formatDate(date, 'GMT+8', 'yyyyMMdd'); const randomNum = Math.floor(Math.random() * 10000).toString().padStart(4, '0'); return `ORD${dateStr}${randomNum}`; } function updateCustomerData(sheet, data) { const lastRow = sheet.getLastRow(); if (lastRow < 2) { sheet.appendRow([ data.name, data.phone, 1, parseFloat(data.totalAmount) || 0, new Date() ]); return; } const customers = sheet.getRange(2, 1, lastRow - 1, 5).getValues(); let found = false; for (let i = 0; i < customers.length; i++) { if (customers[i][1] === data.phone) { const rowNum = i + 2; sheet.getRange(rowNum, 3).setValue(customers[i][2] + 1); sheet.getRange(rowNum, 4).setValue(customers[i][3] + parseFloat(data.totalAmount || 0)); sheet.getRange(rowNum, 5).setValue(new Date()); found = true; break; } } if (!found) { sheet.appendRow([ data.name, data.phone, 1, parseFloat(data.totalAmount) || 0, new Date() ]); } } function sendOrderNotification(orderNumber, data, serviceItemsText) { const subject = `【新訂單通知】${orderNumber} - ${data.serviceType}`; const body = ` 新訂單資訊： 訂單編號：${orderNumber} 服務類型：${data.serviceType} 客戶姓名：${data.name} 聯絡電話：${data.phone} 預約日期：${data.bookingDate} 預約時間：${data.bookingTime} 起點地址：${data.startAddress} 終點地址：${data.endAddress} 服務項目明細： ${serviceItemsText} 其他需求：${data.requirements || '無'} 總計金額：NT$ ${data.totalAmount} 請盡快處理此訂單。 --- 此郵件由系統自動發送 `; MailApp.sendEmail(NOTIFICATION_EMAIL, subject, body); } function doGet(e) { const action = e.parameter.action; if (action === 'getStatistics') { return getStatistics(); } else if (action === 'test') { return ContentService .createTextOutput('API 運作正常！') .setMimeType(ContentService.MimeType.TEXT); } return ContentService .createTextOutput(JSON.stringify({ error: 'Invalid action' })) .setMimeType(ContentService.MimeType.JSON); } function getStatistics() { try { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const orderSheet = ss.getSheetByName('訂單資料'); const costSheet = ss.getSheetByName('成本統計'); const customerSheet = ss.getSheetByName('客戶資料'); const orderLastRow = orderSheet.getLastRow(); const totalOrders = orderLastRow > 1 ? orderLastRow - 1 : 0; let totalRevenue = 0; if (orderLastRow > 1) { const orders = orderSheet.getRange(2, 13, orderLastRow - 1, 1).getValues(); for (let i = 0; i < orders.length; i++) { totalRevenue += parseFloat(orders[i][0]) || 0; } } const costLastRow = costSheet.getLastRow(); let totalCost = 0; let totalProfit = 0; if (costLastRow > 1) { const costs = costSheet.getRange(2, 5, costLastRow - 1, 2).getValues(); for (let i = 0; i < costs.length; i++) { totalCost += parseFloat(costs[i][0]) || 0; totalProfit += parseFloat(costs[i][1]) || 0; } } const customerLastRow = customerSheet.getLastRow(); const totalCustomers = customerLastRow > 1 ? customerLastRow - 1 : 0; return ContentService .createTextOutput(JSON.stringify({ success: true, totalOrders: totalOrders, totalRevenue: totalRevenue, totalCost: totalCost, totalProfit: totalProfit, totalCustomers: totalCustomers })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: error.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
